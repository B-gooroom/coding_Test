```sql
SELECT
  UGU.USER_ID,
  UGU.NICKNAME,
  UGB.TOTAL_SALES AS TOTAL_SALES
FROM USED_GOODS_USER UGU
  INNER JOIN (SELECT
                WRITER_ID,
                STATUS,
                SUM(PRICE) AS TOTAL_SALES
              FROM USED_GOODS_BOARD
              WHERE STATUS = 'DONE'
              GROUP BY WRITER_ID, STATUS
              HAVING SUM(PRICE) >= 700000) UGB
    ON UGU.USER_ID = UGB.WRITER_ID
ORDER BY UGB.TOTAL_SALES
```
